  `cddjlrzb` varchar(255) NULL COMMENT '超大单净流入占比',
  `ddjlr` varchar(255) NULL COMMENT '大单净流入金额',
  `ddjlrzb` varchar(255) NULL COMMENT '大单净流入占比'
